program ::= (statement "\n")*

statement ::= name "{" field* "}"
           | "New_" name "{" field* "}"

field ::= type ":" name "=" value

type ::= "String" | "Int" | "Bool" | "Float" | name

value ::= string | number | "true" | "false" | name

name ::= [a-zA-Z][a-zA-Z0-9_]*

string ::= '"' [^"]* '"'

number ::= "-"? [0-9]+ "." [0-9]+
         | "-"? [0-9]+
