<object> ::= <object_name> "{" <object_contents> "}"

<nested_object> ::= "nested_object_name" "{" <nested_object_contents> "}"

<property> ::= <identifier> ":" <value>

<value> ::= <primitive_value> | "[" <array_values> "]"

<primitive_value> ::= <integer> | <string> | <boolean> | <float>

<integer> ::= "i_INT" ":" <integer_value>

<string> ::= "s_STR" ":" <string_value>

<boolean> ::= "b_BOOL" ":" <boolean_value>

<float> ::= "f_FLOAT" ":" <float_value>

<array_values> ::= <primitive_value> <array_values_tail>

<array_values_tail> ::= "," <primitive_value> <array_values_tail> | ε

<identifier> ::= <identifier_part> <identifier_tail>

<identifier_part> ::= [a-zA-Z_]

<identifier_tail> ::= <identifier_part> | <integer>

<integer_value> ::= [0-9]+

<string_value> ::= "\"" <characters> "\""

<characters> ::= <character> <characters> | ε

<boolean_value> ::= "true" | "false"

<float_value> ::= <integer_value> "." <integer_value>
